cmake_minimum_required(VERSION 3.20.0)
set(projectName "xbwuc_net" )
project(${projectName} C CXX)
set(CMAKE_CXX_STANDARD 17)

#include_directories("../include" )
add_library(${projectName} OBJECT "server.h" "jniImpl.cpp"  "client.cpp" "server.cpp"  )

target_include_directories(${projectName} INTERFACE "." "../include")

)
endif(APPLE)
